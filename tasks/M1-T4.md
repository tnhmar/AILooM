## Objective

Implement the full domain event catalog as the `MemoryEvent` base type plus all
concrete event types. Events are the observability backbone of the system —
they are emitted by use cases and consumed by the `ObserverPort`.
Pure Python only.

# 2. MemoryEvent base (frozen dataclass)
@dataclass(frozen=True)
class MemoryEvent:
    event_id: str = field(default_factory=lambda: str(uuid4()))
    event_type: EventType = field(init=False)
    tenant_id: TenantId
    occurred_at: datetime = field(default_factory=datetime.utcnow)
    correlation_id: Optional[str] = None

# 3. All concrete event classes (frozen dataclasses, extend MemoryEvent)
# Use object.__setattr__(self, "event_type", EventType.XXX) in __post_init__
# for each subclass because frozen=True.

@dataclass(frozen=True)
class MemoryWrittenEvent(MemoryEvent):
    memory_id: MemoryId
    scope: Scope
    sector: MemorySector
    pipeline_status: PipelineStatus
