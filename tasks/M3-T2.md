```python
# 1. LLMClientPort
@runtime_checkable
class LLMClientPort(Protocol):
    async def complete(self, system_prompt: str, user_prompt: str) -> str: ...

# 2. Extraction system prompt constant
EXTRACTION_SYSTEM_PROMPT = """
You are a memory extraction assistant. Given a text, extract structured facts.
Return a JSON array with fields:
  subject_entity_id, predicate, predicate_group, object_value,
  confidence (float 0-1), sector (str).
Return [] if no facts. Return ONLY valid JSON.
"""

# 3. LLMExtractionService
class LLMExtractionService:
    """
    Steps:
    1. Build extraction prompt from record.raw_payload.
    2. Call LLMClientPort.complete().
    3. Parse JSON → list of raw fact dicts.
    4. For each raw fact:
       a. Construct candidate Fact (effective_from=record.recorded_at).
       b. Call _resolve_contradiction(candidate, policy).
       c. Save fact via FactRepositoryPort.
    5. Emit FactsExtractedEvent.
    6. Return ExtractionResult.
    On JSON parse error: return ExtractionResult(error=..., facts=[]).

    _resolve_contradiction:
    - Query active facts on same (entity_id, predicate_group).
    - None found → save as ACTIVE.
    - Found + confidence >= threshold → AUTO_CLOSE: close existing, save candidate ACTIVE.
      Emit ContradictionDetectedEvent.
    - Found + confidence < threshold → save candidate as PROPOSED.
      Emit ContradictionLowConfidenceEvent.
    """

    def __init__(self, llm_client: LLMClientPort, fact_repo: FactRepositoryPort,
                 observer: ObserverPort,
                 policy: Optional[ConflictResolutionPolicy] = None) -> None: ...

    async def extract(self, record: MemoryRecord) -> ExtractionResult: ...

    async def _resolve_contradiction(self, candidate: Fact,
                                      policy: ConflictResolutionPolicy) -> Fact: ...
```

**Generate `tests/engine/test_extraction.py`:**

Tests:
1. Valid JSON with 2 facts → `len(result.facts) == 2`.
2. Empty JSON `[]` → `facts=[]`, `error=None`.
3. Invalid JSON → `error` set, `facts=[]`.
4. `fact_repo.save` called once per extracted fact.
5. `FactsExtractedEvent` emitted via `observer.emit`.
6. High-confidence fact with existing active fact triggers `ContradictionDetectedEvent`.
7. High-confidence auto-close calls `fact_repo.close_fact`.
8. Low-confidence fact triggers `ContradictionLowConfidenceEvent`.
9. Low-confidence fact saved with `lifecycle_state=PROPOSED`.
10. No existing fact → new fact saved as `lifecycle_state=ACTIVE`.
