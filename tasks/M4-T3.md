class SessionEndHandler:
    """
    Implements NotifySessionEndedUseCase.

    Flow:
    1. Count ACTIVE records in the session scope:
       record_repo.list_by_scope(scope, lifecycle_states=[ACTIVE], limit=10_000)
       count = len(records).
    2. Emit SessionEndedEvent(session_id, scope, record_count=count).
    3. Load ConsolidationPolicy via policy_repo.
    4. If policy.trigger == SESSION_END and policy.enabled:
       await consolidation_service.execute(tenant_id, scope=scope).
    5. Return None.
    """

    def __init__(
        self,
        record_repo: MemoryRecordRepositoryPort,
        policy_repo: TenantPolicyRepositoryPort,
        observer: ObserverPort,
        consolidation_service: ConsolidationService,
    ) -> None: ...

    async def execute(
        self,
        tenant_id: TenantId,
        session_id: SessionId,
        scope: Scope,
    ) -> None: ...
