                 policy_repo: TenantPolicyRepositoryPort, observer: ObserverPort,
                 graph_available: bool = False) -> None: ...

# 2. RecallMemoryService
class RecallMemoryService:
    """
    1. Convert RecallRequest → SearchRequest (k = max_items * 2 for headroom).
    2. SearchMemoryService.execute(search_request).
    3. No results → RecallResult(status=NO_MATCH). Emit RecallNoMatchEvent.
    4. Package with token budget:
       - Token estimate: int(len(content.split()) * 1.3)
       - Accumulate until max_tokens or max_items reached.
       - explanation = f"Retrieved via {mode} search, score={score:.3f}"
       - trace_id = new_trace_id() per item.
    5. status=MATCH if full, PARTIAL_MATCH if token-cut.
    6. Emit MemoryRecalledEvent.
    7. Return RecallResult.
    """
    def __init__(self, search_service: SearchMemoryService,
                 observer: ObserverPort) -> None: ...
    async def execute(self, request: RecallRequest) -> RecallResult: ...
    def _estimate_tokens(self, text: str) -> int: ...
