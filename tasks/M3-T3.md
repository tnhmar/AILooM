```python
# 1. IndexTarget
class IndexTarget(str, Enum):
    VECTOR    = "VECTOR"
    FULL_TEXT = "FULL_TEXT"
    GRAPH     = "GRAPH"
    TEMPORAL  = "TEMPORAL"

# 2. QueryPlan
@dataclass
class QueryPlan:
    mode: SearchMode
    targets: List[IndexTarget]
    weights: dict
    use_llm_rerank: bool = False
    temporal_filter: Optional[TemporalFilter] = None
    k_per_index: int = 20
    final_k: int = 10
    explanation: str = ""

# 3. QueryPlannerPort
@runtime_checkable
class QueryPlannerPort(Protocol):
    def plan(self, request: SearchRequest,
             weights: SearchWeightsPolicy,
             graph_available: bool) -> QueryPlan: ...

# 4. DefaultQueryPlanner (mode → plan mapping per ADR-010)
# SEMANTIC → [VECTOR], KEYWORD → [FULL_TEXT],
# HYBRID → [VECTOR, FULL_TEXT], HYBRID_TEMPORAL → [VECTOR, FULL_TEXT, TEMPORAL],
# QUALITY → [VECTOR, FULL_TEXT, TEMPORAL] + optional rerank,
# GRAPH → [GRAPH] or CapabilityNotAvailableError if graph_available=False
# k_per_index = max(final_k * 3, 20)
class DefaultQueryPlanner:
    def plan(self, request: SearchRequest,
             weights: SearchWeightsPolicy,
             graph_available: bool) -> QueryPlan: ...
```
