# 1. ConsolidationTrigger
class ConsolidationTrigger(str, Enum):
    SESSION_END  = "SESSION_END"
    THRESHOLD    = "THRESHOLD"
    SCHEDULE     = "SCHEDULE"
    ON_DEMAND    = "ON_DEMAND"

# 2. ConflictResolutionMode
class ConflictResolutionMode(str, Enum):
    AUTO_CLOSE   = "AUTO_CLOSE"
    FLAG_ONLY    = "FLAG_ONLY"
    MANUAL       = "MANUAL"
    LLM_ARBITRATED = "LLM_ARBITRATED"

# 3. ConsolidationPolicy
@dataclass
class ConsolidationPolicy:
    policy_id: PolicyId = field(default_factory=new_policy_id)
    trigger: ConsolidationTrigger = ConsolidationTrigger.SESSION_END
    threshold_record_count: int = 500
    max_items_per_run: int = 200
    sectors: List[MemorySector] = field(default_factory=lambda: list(MemorySector))
    enabled: bool = True

# 4. RetentionPolicy
@dataclass
class RetentionPolicy:
    policy_id: PolicyId = field(default_factory=new_policy_id)
    decay_after_days: Optional[int] = 90
    archive_after_days: Optional[int] = 365
    delete_after_days: Optional[int] = None
    sector_decay_overrides: dict = field(default_factory=dict)
    enabled: bool = True

# 5. ConflictResolutionPolicy
@dataclass
class ConflictResolutionPolicy:
    policy_id: PolicyId = field(default_factory=new_policy_id)
    mode: ConflictResolutionMode = ConflictResolutionMode.AUTO_CLOSE
    low_confidence_threshold: float = 0.6
    enabled: bool = True

# 6. SearchWeightsPolicy
@dataclass
class SearchWeightsPolicy:
    policy_id: PolicyId = field(default_factory=new_policy_id)
    semantic_weight: float = 0.5
    keyword_weight: float = 0.2
    entity_weight: float = 0.15
    recency_weight: float = 0.1
    salience_weight: float = 0.05
    enabled: bool = True

    def total_weight(self) -> float:
        return (
            self.semantic_weight + self.keyword_weight
            + self.entity_weight + self.recency_weight + self.salience_weight
        )

# 7. TenantPolicies
@dataclass
class TenantPolicies:
    consolidation: ConsolidationPolicy = field(default_factory=ConsolidationPolicy)
    retention: RetentionPolicy = field(default_factory=RetentionPolicy)
    conflict_resolution: ConflictResolutionPolicy = field(default_factory=ConflictResolutionPolicy)
    search_weights: SearchWeightsPolicy = field(default_factory=SearchWeightsPolicy)
